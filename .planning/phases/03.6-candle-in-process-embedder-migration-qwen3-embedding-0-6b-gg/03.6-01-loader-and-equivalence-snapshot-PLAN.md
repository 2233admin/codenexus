use anyhow::{Context, Result};
use std::sync::OnceLock;

// PRESERVED VERBATIM from prior ollama embedder -- byte-for-byte unchanged
// per Pitfall 4 / locked decision A4. The trailing space after "Query:" is
// part of the empirical config that produced poc.db's 67.9% B1-B7 baseline.
const QUERY_INSTRUCT: &str =
    "Instruct: Given a natural language code search query, retrieve the most relevant code symbol from a TypeScript codebase\nQuery: ";

#[derive(Copy, Clone)]
pub enum Role { Query, Passage }

pub struct Embedder {
    inner: OnceLock<fastembed::Qwen3TextEmbedding>,
}

impl Embedder {
    pub fn new() -> Self { Self { inner: OnceLock::new() } }

    fn ensure_loaded(&self) -> Result<&fastembed::Qwen3TextEmbedding> {
        if let Some(m) = self.inner.get() { return Ok(m); }
        let init_opts = fastembed::Qwen3TextInitOptions::default();
        // The crate's default uses Qwen/Qwen3-Embedding-0.6B from HF Hub.
        // Verify dim=1024, f32 output, instruction-prefix configurable.
        let model = fastembed::Qwen3TextEmbedding::try_new(init_opts)
            .context("fastembed Qwen3TextEmbedding init")?;
        let _ = self.inner.set(model);
        Ok(self.inner.get().unwrap())
    }

    pub fn embed(&self, text: &str, role: Role) -> Result<Vec<f32>> {
        const MAX_ATTEMPTS: u32 = 5;
        const BASE_DELAY_MS: u64 = 250;
        let mut last_err: Option<anyhow::Error> = None;
        for attempt in 0..MAX_ATTEMPTS {
            match self.embed_once(text, role) {
                Ok(v) => return Ok(v),
                Err(e) => {
                    last_err = Some(e);
                    if attempt + 1 < MAX_ATTEMPTS {
                        let delay = BASE_DELAY_MS << attempt;
                        std::thread::sleep(std::time::Duration::from_millis(delay));
                    }
                }
            }
        }
        Err(last_err.unwrap())
    }

    fn embed_once(&self, text: &str, role: Role) -> Result<Vec<f32>> {
        let model = self.ensure_loaded()?;
        let prompt: String = match role {
            Role::Query => format!("{}{}", QUERY_INSTRUCT, text),
            Role::Passage => text.to_string(),
        };
        // fastembed Qwen3 batch API: embed(Vec<S>) -> Vec<Vec<f32>>
        // Verify: returned dim is 1024 (NOT 768 / 384 / etc).
        let mut out = model.embed(vec![prompt], None)
            .context("fastembed embed")?;
        let v = out.pop().context("fastembed empty result")?;
        anyhow::ensure!(v.len() == 1024, "expected dim=1024, got {}", v.len());
        Ok(v)
    }
}

pub fn cosine(a: &[f32], b: &[f32]) -> f32 {
    if a.len() != b.len() || a.is_empty() { return 0.0; }
    let mut dot = 0.0f32; let mut na = 0.0f32; let mut nb = 0.0f32;
    for i in 0..a.len() { dot += a[i]*b[i]; na += a[i]*a[i]; nb += b[i]*b[i]; }
    let denom = (na.sqrt() * nb.sqrt()).max(1e-12);
    dot / denom
}

            VarBuilder::from_mmaped_safetensors(&[weights_path], DType::F32, &device)
                .context("VarBuilder mmap")?
        };
        let model = Model::new(&cfg, vb).context("qwen3::Model::new")?;
        let tokenizer = tokenizers::Tokenizer::from_file(tokenizer_path)
            .map_err(|e| anyhow::anyhow!("tokenizer load: {}", e))?;
        let _ = self.state.set(EmbedderState { model: std::sync::Mutex::new(model), tokenizer, device });
        Ok(self.state.get().unwrap())
    }

    pub fn embed(&self, text: &str, role: Role) -> Result<Vec<f32>> {
        // Retry wrapper preserved from Phase 3.5b commit 8f4da66 -- defensive-only per ARCH §9.9 D-W9.
        const MAX_ATTEMPTS: u32 = 5;
        const BASE_DELAY_MS: u64 = 250;
        let mut last_err: Option<anyhow::Error> = None;
        for attempt in 0..MAX_ATTEMPTS {
            match self.embed_once(text, role) {
                Ok(v) => return Ok(v),
                Err(e) => {
                    last_err = Some(e);
                    if attempt + 1 < MAX_ATTEMPTS {
                        let delay = BASE_DELAY_MS << attempt;
                        std::thread::sleep(std::time::Duration::from_millis(delay));
                    }
                }
            }
        }
        Err(last_err.unwrap())
    }

    fn embed_once(&self, text: &str, role: Role) -> Result<Vec<f32>> {
        let st = self.ensure_loaded()?;
        let prompt: String = match role {
            Role::Query => format!("{}{}", QUERY_INSTRUCT, text),
            Role::Passage => text.to_string(),
        };
        let encoding = st.tokenizer.encode(prompt, true)
            .map_err(|e| anyhow::anyhow!("tokenize: {}", e))?;
        let ids: Vec<u32> = encoding.get_ids().to_vec();
        let input_ids = Tensor::new(&ids[..], &st.device)?.unsqueeze(0)?;
        let mut model = st.model.lock().unwrap();
        let last_hidden = model.forward(&input_ids, 0)?;  // (1, seq_len, 1024)
        let (_b, seq, _h) = last_hidden.dims3()?;
        let pooled = last_hidden.narrow(1, seq - 1, 1)?
                                .squeeze(1)?
                                .squeeze(0)?;
        let pooled_f32: Vec<f32> = pooled.to_vec1()?;
        let norm = pooled_f32.iter().map(|x| x*x).sum::<f32>().sqrt().max(1e-12);
        let normalized: Vec<f32> = pooled_f32.into_iter().map(|x| x / norm).collect();
        anyhow::ensure!(normalized.len() == 1024, "expected dim=1024 got {}", normalized.len());
        Ok(normalized)
    }
}

// Source-of-truth version-hash compute binary for ARCH §9.8.
//
// Rationale: bash `echo -n "...\nQuery: "` on POSIX / git-bash emits a
// LITERAL backslash-n (2 bytes), NOT a real LF (1 byte 0x0A). The Rust
// const QUERY_INSTRUCT in embedder.rs is a Rust string literal where
// `\n` is parsed at compile time into a real LF. If §9.8 hash were
// computed via bash echo, the committed hash would never match runtime
// behavior, breaking drift detection silently from day one.
//
// This binary imports QUERY_INSTRUCT from embedder.rs (single source of
// truth) and prints sha256(model_id|dim|QUERY_INSTRUCT)[:12] hex.
// Plan 2 Task 4.2 pipes this output into ARCH §9.8 history table.

use poc_retrieval::embedder::QUERY_INSTRUCT;
use sha2::{Digest, Sha256};

const MODEL_ID: &str = "Qwen/Qwen3-Embedding-0.6B";
const DIM: u32 = 1024;

fn main() {
    let payload = format!("{}|{}|{}", MODEL_ID, DIM, QUERY_INSTRUCT);
    let mut hasher = Sha256::new();
    hasher.update(payload.as_bytes());
    let digest = hasher.finalize();
    let hex: String = digest.iter().take(6).map(|b| format!("{:02x}", b)).collect();
    // 6 bytes * 2 hex chars = 12-char prefix per ARCH §9.8 convention.
    println!("{}", hex);
}

Cmd::EmbedEquivalence { queries, ollama_snapshot, out } => {
    let raw_q = std::fs::read_to_string(&queries)?;
    let qs: Vec<EvalQuery> = serde_json::from_str(&raw_q)?;
    let raw_s = std::fs::read_to_string(&ollama_snapshot)?;
    let snap: Vec<serde_json::Value> = serde_json::from_str(&raw_s)?;
    // Index snapshot by (id, role) -> Vec<f32>
    let mut snap_idx: std::collections::HashMap<(String, String), Vec<f32>> = std::collections::HashMap::new();
    for entry in &snap {
        let id = entry["id"].as_str().unwrap().to_string();
        let role = entry["role"].as_str().unwrap().to_string();
        let vec: Vec<f32> = entry["vec"].as_array().unwrap()
            .iter().map(|v| v.as_f64().unwrap() as f32).collect();
        snap_idx.insert((id, role), vec);
    }
    let embedder = embedder::Embedder::new();
    let mut per_query = Vec::new();
    let mut cosines = Vec::new();
    for q in &qs {
        for role_str in &["Query", "Passage"] {
            let role = if *role_str == "Query" { embedder::Role::Query } else { embedder::Role::Passage };
            let candle_vec = embedder.embed(&q.query, role)?;
            let ollama_vec = snap_idx.get(&(q.id.clone(), role_str.to_string()))
                .ok_or_else(|| anyhow::anyhow!("missing snapshot entry for {}/{}", q.id, role_str))?;
            let c = embedder::cosine(&candle_vec, ollama_vec);
            per_query.push(serde_json::json!({"id": q.id, "role": role_str, "cosine": c}));
            cosines.push(c);
        }
    }
    cosines.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let n = cosines.len() as f32;
    let mean: f32 = cosines.iter().sum::<f32>() / n;
    let pct = |p: f32| -> f32 { let i = ((p/100.0) * n).floor() as usize; cosines[i.min(cosines.len()-1)] };
    let p10 = pct(10.0); let p50 = pct(50.0); let p90 = pct(90.0);
    let summary = serde_json::json!({
        "n": cosines.len(),
        "mean_cosine": mean,
        "p10_cosine": p10,
        "p50_cosine": p50,
        "p90_cosine": p90,
        "passes_gate": mean >= 0.97 && p10 >= 0.95,
        "per_query": per_query,
    });
    std::fs::create_dir_all(out.parent().unwrap_or(std::path::Path::new(".")))?;
    std::fs::write(&out, serde_json::to_string_pretty(&summary)?)?;
    eprintln!("=== Equivalence summary ===");
    eprintln!("n={}  mean={:.4}  p10={:.4}  p50={:.4}  p90={:.4}", cosines.len(), mean, p10, p50, p90);
    eprintln!("gate (mean>=0.97 AND p10>=0.95): {}", if mean >= 0.97 && p10 >= 0.95 { "PASS" } else { "FAIL" });
    if !(mean >= 0.97 && p10 >= 0.95) {
        anyhow::bail!("equivalence gate failed: mean={:.4} p10={:.4}", mean, p10);
    }
}
