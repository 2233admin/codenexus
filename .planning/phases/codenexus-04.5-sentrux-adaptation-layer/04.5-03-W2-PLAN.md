//! AliasIndex -- read-only projection of `&[AliasDecl]` (CONTEXT.md lines 72-77).
//!
//! Built once per `resolve()` call from the AliasDecls of a single file
//! (or a slice covering multiple files for global lookups). Pure
//! function: same input always yields the same index. Internal scaffold,
//! not persisted.
//!
//! AliasDecl is consumed from `codenexus_types` (UQ1 W1 resolution); not
//! `codenexus_core::types` (which is a re-export shim during transition).

use std::collections::HashMap;
use codenexus_types::AliasDecl;

/// Composite-key index over (file, alias) -> AliasDecl. Construction is
/// O(N) in the number of decls; lookup is O(1).
#[derive(Debug, Clone)]
pub struct AliasIndex {
    by_file_alias: HashMap<(String, String), AliasDecl>,
}

impl AliasIndex {
    /// Build from a slice of AliasDecls. If two decls share (file, alias),
    /// the FIRST (in iteration order) wins; subsequent duplicates are
    /// dropped silently. This matches the Store::insert_alias_decl
    /// `INSERT OR IGNORE` semantics.
    pub fn from_decls(decls: &[AliasDecl]) -> Self {
        let mut by_file_alias = HashMap::with_capacity(decls.len());
        for d in decls {
            by_file_alias
                .entry((d.from_file.clone(), d.alias.clone()))
                .or_insert_with(|| d.clone());
        }
        Self { by_file_alias }
    }

    /// Look up an AliasDecl by (file, alias). Returns None if no decl matches.
    pub fn lookup(&self, file: &str, alias: &str) -> Option<&AliasDecl> {
        self.by_file_alias.get(&(file.to_string(), alias.to_string()))
    }

    /// Total number of AliasDecls indexed (for diagnostics / coverage tests).
    pub fn len(&self) -> usize {
        self.by_file_alias.len()
    }

    pub fn is_empty(&self) -> bool {
        self.by_file_alias.is_empty()
    }
}

#[cfg(test)]
mod tests {

    fn d(from: &str, alias: &str, target: &str, member: Option<&str>) -> AliasDecl {
        AliasDecl {
            from_file: from.to_string(),

            target_file: target.to_string(),
            target_member: member.map(String::from),

    }

//! SymbolIndex -- read-only projection of `Store` (CONTEXT.md lines 66-71).
//!
//! Provides per-file and globally-unique name lookups consumed by the
//! resolver. Built ONCE per `resolve()` call (not per call site), so
//! repeated resolve() invocations during a full re-index amortize the
//! Store traversal across thousands of call sites.

use std::collections::HashMap;
use anyhow::Result;
use codenexus_core::storage::Store;
// Note: codenexus_core::storage::Store is consumed for build-time projection;
// the Symbol type itself comes from codenexus_types per UQ1 W1 resolution.

/// Cached projection: per-file (name -> id) maps + global (name -> Vec<id>) map.
#[derive(Debug)]
pub struct SymbolIndex {
    /// HashMap<file, HashMap<name, id>>. id is rowid in symbols table.
    files: HashMap<String, HashMap<String, i64>>,
    /// Global name index. Vec<id> because the same name can appear in
    /// multiple files; `find_global_unique` returns Some only when the
    /// vec has exactly 1 entry.
    by_name_global: HashMap<String, Vec<i64>>,
}

impl SymbolIndex {
    /// Build the index by traversing all symbols in the Store. O(total
    /// symbols) -- ~2300 symbols for fsc.db; under a second on the dev
    /// machine.
    pub fn from_store(store: &Store) -> Result<Self> {
        let mut files: HashMap<String, HashMap<String, i64>> = HashMap::new();
        let mut by_name_global: HashMap<String, Vec<i64>> = HashMap::new();

        let file_paths = store.list_files()?;
        for f in &file_paths {
            let syms = store.symbols_in_file_full(f)?;
            // symbols_in_file_full returns Vec<(id, name, start_line, end_line)>
            let entry = files.entry(f.clone()).or_default();
            for (id, name, _sl, _el) in syms {
                entry.entry(name.clone()).or_insert(id);
                by_name_global.entry(name).or_default().push(id);
            }
        }

        Ok(Self { files, by_name_global })
    }

    /// Find a symbol by (file, name). Returns the FIRST matching id (per-file
    /// duplicates are unusual; first-wins matches the existing
    /// Store::symbol_in_file_by_name semantics).
    pub fn find_in_file(&self, file: &str, name: &str) -> Option<i64> {
        self.files.get(file)?.get(name).copied()
    }

    /// Find a globally-unique symbol by name. Returns Some(id) ONLY if
    /// exactly one symbol in the entire index has this name. Multiple
    /// matches return None (resolver falls through; ambiguous matches
    /// don't false-resolve).
    pub fn find_global_unique(&self, name: &str) -> Option<i64> {
        let v = self.by_name_global.get(name)?;
        if v.len() == 1 { Some(v[0]) } else { None }
    }

    /// Count of indexed files (for tests).
    pub fn file_count(&self) -> usize {
        self.files.len()
    }

    /// Count of indexed (file, name) pairs (for tests).
    pub fn total_symbols(&self) -> usize {
        self.files.values().map(|h| h.len()).sum()
    }
}

//! TOML-data-driven via `RepoCtx::language_semantics`. Cherry-picked from

impl<'a> PathResolver<'a> {
    pub fn new(repo_ctx: &'a RepoCtx) -> Self {
        Self { repo_ctx }
    }

    /// Resolve an import source string from the perspective of `from_file`
    /// to a repo-relative file path. Returns None for bare/external
    /// imports or when no candidate file exists in the repo.
    ///
    /// Resolution order:
    ///   1. tsconfig path-alias (e.g. `@/X` -> `src/X`)
    ///   2. relative path (`./X` / `../X`)
    ///   3. source_root-rooted (e.g. `X` from a tsconfig source_root config)
    ///   4. Bare imports always None.
    ///
    /// Returns None when no candidate exists on disk under repo_root.
    pub fn resolve(&self, from_file: &str, source: &str) -> Option<String> {
        // Step 1: tsconfig path alias
        if let Some(resolved) = self.resolve_path_alias(source) {
            if let Some(found) = self.find_with_extensions(&resolved) {
                return Some(found);
            }
        }

        // Step 2: relative path
        if source.starts_with('.') {
            return self.resolve_relative(from_file, source);
        }

        // Step 3: source_root fallback (if config has source_root configured)
        // ... (lifted from sentrux source_root walk logic)

        // Step 4: bare import -> None
        None
    }

    fn resolve_path_alias(&self, source: &str) -> Option<String> {
        for (pattern, target) in &self.repo_ctx.ts_path_aliases {
            // Sentrux suffix.rs has the glob/wildcard logic here; replicate
            // verbatim. Standard TS alias is `@/*` -> `src/*` and similar
            // wildcard substitution.
            if let Some(rest) = strip_alias_pattern(source, pattern) {
                let resolved = target.replace("*", &rest);
                return Some(resolved);
            }
        }
        None
    }

    fn resolve_relative(&self, from_file: &str, source: &str) -> Option<String> {
        // Replicates graph_build.rs::resolve_import_path (lines 487-536) byte-equivalently:
        // parent-dir + source -> normalize -> .js->.ts strip -> extension matrix walk.
        let from_norm = from_file.replace('\\', "/");
        let dir = match from_norm.rsplit_once('/') {
            Some((d, _)) => d,
            None => "",
        };
        let combined = if dir.is_empty() {
            source.to_string()
        } else {
            format!("{}/{}", dir, source)
        };
        let normalised = normalize_relative_path(&combined);

        // TS .js -> .ts strip
        let stripped = normalised
            .strip_suffix(".js")
            .map(String::from)
            .or_else(|| normalised.strip_suffix(".jsx").map(String::from))
            .unwrap_or_else(|| normalised.clone());

        for base in &[stripped.as_str(), normalised.as_str()] {
            if let Some(found) = self.find_with_extensions(base) {
                return Some(found);
            }
        }
        None
    }

    /// Walk the resolve_extensions matrix and return the first hit on disk.
    /// Includes Windows backslash variant (graph_build.rs line 525-526).
    fn find_with_extensions(&self, base: &str) -> Option<String> {
        let extensions = &["", ".ts", ".tsx", "/index.ts", "/index.tsx"];
        for suffix in extensions {
            let candidate = format!("{}{}", base, suffix);
            for c in &[candidate.clone(), candidate.replace('/', "\\")] {
                let abs = self.repo_ctx.repo_root.join(c);
                if abs.exists() {
                    return Some(c.clone());
                }
            }
        }
        None
    }
}

/// Strip an alias pattern prefix from a source string. e.g. for pattern
/// `@/*` and source `@/foo/bar`, returns `Some("foo/bar")`.
fn strip_alias_pattern(source: &str, pattern: &str) -> Option<String> {
    if let Some(prefix) = pattern.strip_suffix("/*") {
        if let Some(rest) = source.strip_prefix(prefix) {
            let rest = rest.strip_prefix('/').unwrap_or(rest);
            return Some(rest.to_string());
        }
    }
    if pattern == source {
        return Some(String::new());
    }
    None
}

/// Normalize `a/b/./c/../d` -> `a/b/d`. Forward-slash only. Drops leading
/// `./` segments. Verbatim from graph_build.rs::normalize_relative_path
/// (lines 562-574) -- shared utility moved here.
fn normalize_relative_path(p: &str) -> String {
    let mut out: Vec<&str> = Vec::new();
    for seg in p.split('/') {
        match seg {
            "" | "." => continue,
            ".." => { out.pop(); }
            s => out.push(s),
        }
    }
    out.join("/")
}

/// Parse `tsconfig.json` `compilerOptions.paths` into a flat HashMap.
/// Returns empty map if file absent. Wired into RepoCtx::prepare via
/// the `parse_ts_path_aliases` function.
pub fn parse_ts_path_aliases(repo_root: &Path) -> Result<HashMap<String, String>> {
    let tsconfig = repo_root.join("tsconfig.json");
    if !tsconfig.exists() {
        return Ok(HashMap::new());
    }
    let bytes = std::fs::read(&tsconfig)?;
    let value: serde_json::Value = match serde_json::from_slice(&bytes) {
        Ok(v) => v,
        Err(_) => return Ok(HashMap::new()),  // tsconfig may have JSON5 comments; tolerate
    };
    let paths = value
        .get("compilerOptions")
        .and_then(|c| c.get("paths"))
        .and_then(|p| p.as_object());
    let mut out = HashMap::new();
    if let Some(paths) = paths {
        for (alias, targets) in paths {
            if let Some(targets) = targets.as_array() {
                if let Some(first) = targets.first().and_then(|t| t.as_str()) {
                    out.insert(alias.clone(), first.to_string());
                }
            }
        }
    }
    Ok(out)
}

    ///          target_file, MISSES, falls through to step 3 (global-unique
    ///          on `bar`, also misses) -> returns None. Pinned bug preserved.
    ///   3. GlobalUnique: SymbolIndex::find_global_unique(name)
    pub fn resolve(
        from_file: &str,
        name: &str,
        ns_obj: Option<&str>,
        symbols: &SymbolIndex,
        aliases: &AliasIndex,
    ) -> Option<(i64, ResolutionMethod)> {
        // Step 1: SameFile -- only if no ns_obj (X.foo() never resolves
        // same-file because X is by-construction a different file's namespace)
        if ns_obj.is_none() {
            if let Some(id) = symbols.find_in_file(from_file, name) {
                return Some((id, ResolutionMethod::SameFile));
            }
        }

        // Step 2: ImportResolved -- branch on ns_obj
        if let Some(obj) = ns_obj {
            // Namespace path: look up the namespace alias for (file, obj)
            if let Some(decl) = aliases.lookup(from_file, obj) {
                if decl.target_member.is_none() {
                    // True namespace import (`import * as X`): look up
                    // `name` directly in target_file
                    if let Some(id) = symbols.find_in_file(&decl.target_file, name) {
                        return Some((id, ResolutionMethod::ImportResolved));
                    }
                }
            }
            // ns_obj path either missed or AliasDecl was not namespace-shaped.
            // Fall through to GlobalUnique on `name` (preserves
            // resolve_with_namespace fallthrough at graph_build.rs line 482).
        } else {
            // Direct path: look up `name` in AliasIndex
            if let Some(decl) = aliases.lookup(from_file, name) {
                // Look up `name` (the local binding) in target_file
                if let Some(id) = symbols.find_in_file(&decl.target_file, name) {
                    return Some((id, ResolutionMethod::ImportResolved));
                }
                // T3 case: target_file does NOT have `name` (because it
                // exports under the original target_member name, not the
                // local binding). Fall through to step 3.
                //
                // FUTURE T3 fix would here add: if decl.target_member is
                // Some AND name != target_member, try lookup of
                // target_member in target_file. NOT done in 04.5-03.
                let _ = decl;  // silence unused warning if no future fix
            }
        }

        // Step 3: GlobalUnique
        if let Some(id) = symbols.find_global_unique(name) {
            return Some((id, ResolutionMethod::GlobalUnique));
        }

        None
    }
}
