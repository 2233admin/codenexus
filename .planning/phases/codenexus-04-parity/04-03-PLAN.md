1. Create `experiments/poc-retrieval/eval/e2e_first_run_smoke.sh` as an executable bash script. Contents:

   ```bash
   #!/usr/bin/env bash
   # E2E first-run smoke harness for Phase 4 first slice (Q6=B locked).
   # Covers acceptance gates that require runtime evidence:
   #   R1.c reload test  : pinned SHA reborn after cache delete
   #   R2 messaging      : start prompt has URL+ETA, failure path links recovery doc
   #   R3 recovery link  : failure message contains 'embedder-offline-bootstrap'
   #   E2E (1)-(6)       : SPEC Acceptance Criteria E2E smoke section
   #
   # NOT covered here (run separately by verifier):
   #   R4.b synthetic A2A index fail-cluster  : needs panic-injection scaffolding
   #   R5.b synthetic Query single-fail timing : needs panic-injection scaffolding
   # Both R4.b and R5.b verifier-runnable per SPEC; treat as P2 follow-up if
   # the panic-injection harness needs more than 1hr to wire.

   set -u   # NOT -e — we want to capture exit codes for failure-path phases
   set -o pipefail

   REPO_ROOT="$(git rev-parse --show-toplevel)"
   POC_DIR="$REPO_ROOT/experiments/poc-retrieval"
   BIN="$REPO_ROOT/bin/codenexus"
   HF_CACHE_DIR="$HOME/.cache/huggingface/hub/models--Qwen--Qwen3-Embedding-0.6B"
   PINNED_SHA="97b0c614be4d77ee51c0cef4e5f07c00f9eb65b3"
   TEST_DB="$POC_DIR/eval/e2e_smoke.db"
   TEST_REPO="$POC_DIR/fixtures/small-ts-corpus"
   LOG="$POC_DIR/eval/e2e_first_run_smoke.log"

   # Idempotent cleanup at exit (covers HTTPS_PROXY leak — feedback rule)
   trap 'unset HTTPS_PROXY; rm -f "$TEST_DB"' EXIT

   echo "==[ Phase 4 E2E smoke harness ]==" | tee "$LOG"
   echo "Pinned SHA: $PINNED_SHA" | tee -a "$LOG"
   date | tee -a "$LOG"

   # ---- Phase 0: ensure release artifact exists ----
   if [ ! -x "$BIN" ]; then
     echo "[setup] $BIN missing — running 'make build'" | tee -a "$LOG"
     ( cd "$REPO_ROOT" && make build ) >> "$LOG" 2>&1 || {
       echo "FAIL: make build" | tee -a "$LOG"
       exit 10
     }
   fi
   if [ ! -x "$BIN" ]; then
     echo "FAIL: $BIN still missing after make build" | tee -a "$LOG"
     exit 11
   fi

   # ---- Phase 0b: ensure a small pre-indexed test DB exists ----
   # Use whatever small TS corpus the project already has. If
   # fixtures/small-ts-corpus is absent, fall back to indexing this repo's
   # own .planning/ markdown via TS-only treatment skipping (parser will
   # produce 0 symbols which is fine for a load-only smoke; query results
   # will be empty but the embedder cold-start path still exercises).
   if [ ! -d "$TEST_REPO" ]; then
     echo "[setup] $TEST_REPO missing — using $POC_DIR as fallback corpus" | tee -a "$LOG"
     TEST_REPO="$POC_DIR"
   fi

   # Index against the existing cache (uses currently-cached snapshot, no
   # network). This warms TEST_DB so subsequent query phases have data.
   echo "[setup] indexing $TEST_REPO into $TEST_DB (uses cached snapshot)" | tee -a "$LOG"
   "$BIN" index --repo "$TEST_REPO" --db "$TEST_DB" --max-consecutive-fail 5 \
     >> "$LOG" 2>&1 || {
       echo "FAIL: pre-index step" | tee -a "$LOG"
       exit 12
     }

   # ---- E2E Phase 1: clean-cache success path ----
   echo "" | tee -a "$LOG"
   echo "==[ E2E (1)-(3): clean cache + working network ]==" | tee -a "$LOG"
   if [ -d "$HF_CACHE_DIR/snapshots/$PINNED_SHA" ]; then
     echo "[phase 1] removing snapshot $PINNED_SHA to force fresh download" | tee -a "$LOG"
     rm -rf "$HF_CACHE_DIR/snapshots/$PINNED_SHA"
     # Also clear blobs to actually re-download — symlinks resolve back
     # to blobs/ which survive snapshot deletion. For true reload test:
     rm -rf "$HF_CACHE_DIR/blobs"
   fi

   PHASE1_OUT="$POC_DIR/eval/.e2e_phase1.txt"
   "$BIN" query "any text" --db "$TEST_DB" > "$PHASE1_OUT" 2>&1
   PHASE1_EXIT=$?

   echo "[phase 1] exit=$PHASE1_EXIT" | tee -a "$LOG"
   cat "$PHASE1_OUT" | tee -a "$LOG"

   # E2E (1): start prompt with URL + ETA appears
   if grep -qE 'first-run download.*huggingface\.co' "$PHASE1_OUT" \
      && grep -qE '30-60s|broadband' "$PHASE1_OUT"; then
     echo "PASS: E2E (1) start prompt URL+ETA visible" | tee -a "$LOG"
   else
     echo "FAIL: E2E (1) start prompt missing URL or ETA wording" | tee -a "$LOG"
     exit 21
   fi

   # E2E (2): download completes — proxy: snapshot dir reborn at pinned SHA
   if [ -d "$HF_CACHE_DIR/snapshots/$PINNED_SHA" ]; then
     echo "PASS: E2E (2) snapshot dir reborn at pinned SHA = R1.c reload test" | tee -a "$LOG"
   else
     echo "FAIL: E2E (2) snapshot dir not reborn at $PINNED_SHA" | tee -a "$LOG"
     ls -la "$HF_CACHE_DIR/snapshots/" | tee -a "$LOG"
     exit 22
   fi

   # E2E (3): query exits 0
   if [ "$PHASE1_EXIT" -eq 0 ]; then
     echo "PASS: E2E (3) query exit 0" | tee -a "$LOG"
   else
     echo "FAIL: E2E (3) query exit $PHASE1_EXIT" | tee -a "$LOG"
     exit 23
   fi

   # ---- E2E Phase 2: network-blocked failure path ----
   echo "" | tee -a "$LOG"
   echo "==[ E2E (4)-(5): network blocked via HTTPS_PROXY ]==" | tee -a "$LOG"
   echo "[phase 2] removing snapshot to force fresh download attempt" | tee -a "$LOG"
   rm -rf "$HF_CACHE_DIR/snapshots/$PINNED_SHA"
   rm -rf "$HF_CACHE_DIR/blobs"

   PHASE2_OUT="$POC_DIR/eval/.e2e_phase2.txt"
   HTTPS_PROXY=http://0.0.0.0:1 HTTP_PROXY=http://0.0.0.0:1 \
     "$BIN" query "any text" --db "$TEST_DB" > "$PHASE2_OUT" 2>&1
   PHASE2_EXIT=$?

   echo "[phase 2] exit=$PHASE2_EXIT" | tee -a "$LOG"
   cat "$PHASE2_OUT" | tee -a "$LOG"

   # E2E (4): failure message contains link to docs/embedder-offline-bootstrap.md
   if grep -qF 'embedder-offline-bootstrap' "$PHASE2_OUT"; then
     echo "PASS: E2E (4) failure message links to recovery doc" | tee -a "$LOG"
   else
     echo "FAIL: E2E (4) failure message missing recovery doc link" | tee -a "$LOG"
     exit 24
   fi

   # E2E (5): exit non-zero
   if [ "$PHASE2_EXIT" -ne 0 ]; then
     echo "PASS: E2E (5) exit non-zero ($PHASE2_EXIT)" | tee -a "$LOG"
   else
     echo "FAIL: E2E (5) expected non-zero exit, got 0" | tee -a "$LOG"
     exit 25
   fi

   # ---- E2E Phase 3: network restored, re-success ----
   echo "" | tee -a "$LOG"
   echo "==[ E2E (6): network restored, re-success ]==" | tee -a "$LOG"
   unset HTTPS_PROXY HTTP_PROXY

   PHASE3_OUT="$POC_DIR/eval/.e2e_phase3.txt"
   "$BIN" query "any text" --db "$TEST_DB" > "$PHASE3_OUT" 2>&1
   PHASE3_EXIT=$?

   echo "[phase 3] exit=$PHASE3_EXIT" | tee -a "$LOG"
   cat "$PHASE3_OUT" | tee -a "$LOG"

   if [ "$PHASE3_EXIT" -eq 0 ] \
      && [ -d "$HF_CACHE_DIR/snapshots/$PINNED_SHA" ] \
      && grep -qE 'first-run download' "$PHASE3_OUT"; then
     echo "PASS: E2E (6) restored network re-succeeds with same SHA reborn" | tee -a "$LOG"
   else
     echo "FAIL: E2E (6)" | tee -a "$LOG"
     exit 26
   fi

   echo "" | tee -a "$LOG"
   echo "==[ ALL E2E PHASES PASSED ]==" | tee -a "$LOG"
   rm -f "$PHASE1_OUT" "$PHASE2_OUT" "$PHASE3_OUT"
   exit 0
   ```
